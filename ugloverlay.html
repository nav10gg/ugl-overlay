    <script>

        // --- 1. PASTE YOUR PUBLISHED GOOGLE SHEETS CSV LINK HERE ---

        const csvUrl = "YOUR_GOOGLE_SHEETS_CSV_LINK_HERE";

        

        // This grabs the Discord ID from the URL (e.g., ?user=123456789)

        const urlParams = new URLSearchParams(window.location.search);

        const playerDiscordId = urlParams.get('user');


        // Helper function to properly read CSV lines

        function parseCSVRow(row) {

            return row.split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/).map(item => item.replace(/(^"|"$)/g, '').trim());

        }


        async function fetchLiveScore() {

            if (!playerDiscordId) return; // If there's no Discord ID in the link, do nothing


            try {

                const response = await fetch(csvUrl);

                const data = await response.text();

                

                // Split the CSV into rows, ignoring the header row

                const rows = data.split('\n').slice(1); 

                let playerIsLive = false;


                for (let row of rows) {

                    if (!row.trim()) continue; // Skip blank lines

                    

                    const columns = parseCSVRow(row);

                    

                    // Column Map based on our exact Stream_Feed:

                    // [2] Game Mode | [3] Total Score 

                    // [4] P1 ID | [5] P2 ID | [6] P3 ID | [7] P4 ID

                    // [8] Total Pot


                    const p1 = columns[4];

                    const p2 = columns[5];

                    const p3 = columns[6];

                    const p4 = columns[7];


                    // Check if our player's Discord ID is in any of the 4 player columns

                    if (playerDiscordId === p1 || playerDiscordId === p2 || playerDiscordId === p3 || playerDiscordId === p4) {

                        

                        // We found them! Update the text on screen.

                        document.getElementById('game-mode').innerText = columns[2] + " RUN";

                        document.getElementById('team-score').innerText = columns[3] || "0";

                        document.getElementById('total-pot').innerText = columns[8] || "$0";

                        

                        // Make sure the box is visible

                        document.getElementById('overlay-box').classList.remove('hidden');

                        playerIsLive = true;

                        break; // Stop searching once we find them

                    }

                }


                // If they aren't in the sheet as LIVE, hide the overlay

                if (!playerIsLive) {

                    document.getElementById('overlay-box').classList.add('hidden');

                }


            } catch (error) {

                console.error("Error fetching data: ", error);

            }

        }


        // Run the fetch immediately, then check again every 10 seconds

        fetchLiveScore();

        setInterval(fetchLiveScore, 10000); 


    </script>
